% ***** ***** ***** PREAMBOLO ***** ***** ***** Inizio
% Il preambolo definisce:
% - il tipo di documento che stai scrivendo
% - il linguaggio che usi
% - i pacchetti di librerie che vorresti usare
% - alcune configurazioni iniziali

% Qui viene configurata:
% - caratteri alti 10 punti
% - pagina A4
% - la tipologia di documento di partenza (scrlttr2)
\documentclass[10pt, a4paper]{scrlttr2}

% inputenc è un pacchetto opzionale ma utile per definire la codifica (nel nostro caso UTF-8)
\usepackage[utf8]{inputenc}

% geometry è un pacchetto che aiuta a gestire le geometrie nel documento. Nel nostro caso configuriamo i bordi
\usepackage[paper=a4paper,right=2cm,left=7.75cm,top=1.7cm,bottom=3cm]{geometry}

% graphicx è un paccheto che useremo per la gestione delle immagini
\usepackage{graphicx}
\graphicspath{ {images/} }

% le seguenti righe gestiranno il font 
\usepackage[defaultfam,light,tabular]{montserrat} %% Option 'defaultfam'
\usepackage{fontspec}
\setsansfont{Montserrat}[
    Path=./MontserratFontFiles/,
    Scale=0.95,
    Extension = .ttf,
    UprightFont=*-Light,
    BoldFont=*-SemiBold,
    ItalicFont=*-LightItalic,
    BoldItalicFont=*-SemiBoldItalic
    ]

% fancyhdr è un pacchetto usato per gestire intestazione e piè di pagina (nel nostro caso useremo lo stile fancy perché personabilizzabile
\usepackage{fancyhdr}
\pagestyle{fancy}
% questo comando pulisce intestazione e piè di pagina dalle configurazioni di default
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
%% Si può aggiungere intestazioni e pié di pagina personalizzati con rispettivamente i comandi \fancyhead e \fancyfoot. A questi, tra parentesi rettangolare, si possono aggiungere dei selettori che permettono di configurare una specifica zona:
% - E per indicare pagine pari (even page)
% - O per indicare pagine dispari (odd page)
% - L per indicare la zona sinistra (left side)
% - C per indicare la zona centrale (centered side)
% - R for indicare la zona destra (right side)
% Nel nostro caso mettiamo il numero di pagina (\thepage) nel piè di pagina a destra [R]
\fancyfoot[R]{\thepage}

% Questa configurazione permette di cambiare la larghezza del testo tra la prima e le pagine successive
\usepackage{everyshi}
\AtNextShipout{\global\leftskip=-4.25cm\relax}

% queste configurazioni permettono di gestire la barra laterale
\usepackage{tikz}
\usepackage{atbegshi}
\usepackage{textpos}
% Barra laterale delle pagine successive alla prima
\TPGrid{8}{11}
\AtBeginShipout{%
    \begin{textblock}{3}(-2.15,0)
        \includegraphics[width=.14\textwidth]{leftBar}
    \end{textblock}
}
% Barra laterale informativa della prima pagina (con i loghi ed info)
\AtBeginShipoutFirst{

    \begin{tikzpicture}[remember picture,overlay]
        \node[anchor=north west,yshift=-1.5cm,xshift=2cm]
        at (current page.north west)
        {
            \includegraphics[width=.48\textwidth]{brancae}
        };
    \end{tikzpicture}
}

% Rimuove la rientranza ad inizio del paragrafo (indentazione)
\setlength{\parindent}{0pt}
